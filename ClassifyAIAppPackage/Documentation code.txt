1. Training / Fine-tuning

from ultralytics import YOLO
import matplotlib
matplotlib.use("Agg")
import matplotlib.pyplot as plt
import pandas as pd
import numpy as np
import shutil
import yaml

# Load model
model = YOLO("yolov8s-cls.pt")  # or your selected .pt file

# Define callback for per-epoch logging
def on_train_epoch_end(trainer):
    metrics = trainer.metrics
    epoch = trainer.epoch + 1
    total = trainer.epochs
    print(f"  EPOCH {epoch}/{total} RESULTS")
    for k, v in metrics.items():
        if v is not None:
            print(f"  {k}: {round(float(v), 4)}")

model.add_callback("on_train_epoch_end", on_train_epoch_end)

# Run training
results = model.train(
    data="/path/to/split/dataset",
    epochs=100,
    imgsz=224,
    batch=16,
    device="0",           # or "cpu"
    workers=8,
    project="/save/folder/AI_Trash_Training",
    name="exp",
    exist_ok=False,
)
